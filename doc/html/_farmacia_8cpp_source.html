119int Farmacia::comprarMedicam(int id_num, int n, PaMedicamento*& result) {
120 int stock_inicial = buscaMedicamID(id_num);
121 if (stock_inicial >= 1) {
122 Stock st(id_num);
123 std::set<Stock>::iterator it = order.find(st);
124 if (it != order.end()) {
125 Stock copy = *it;
126 order.erase(it);
127 copy.decrementa(1); // User buys 1 unit
128 result = copy.getPaMedicamento();
129 order.insert(copy);
130 return stock_inicial;
131 } else {
132 result = nullptr;
133 return stock_inicial;
134 }
135 } else {
136 pedidoMedicam(id_num, n); // Request replenishment
137 result = nullptr;
138 return stock_inicial;
139 }
