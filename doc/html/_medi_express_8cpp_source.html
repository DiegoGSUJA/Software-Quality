206void MediExpress::enlaza_Farmacia_medicamento(std::vector<std::string>& cifs_farmacias) {
207 int numCIFs = static_cast<int>(cifs_farmacias.size());
208 int numMedicamentos = static_cast<int>(medication.size());
209 const int medicamentosPorFarmacia = 100;
210
211 std::vector<int> claves_medicamentos;
212 for (auto const& [key, val] : medication) {
213 claves_medicamentos.push_back(key);
214 }
215
216 for (int i = 0; i < numCIFs; i++) {
217 Farmacia* farmacia = buscarFarmacia(cifs_farmacias[i]);
218 if (farmacia == nullptr) continue;
219
220 for (int j = 0; j < medicamentosPorFarmacia; j++) {
221 int indice_medicamento = (i * medicamentosPorFarmacia + j) % numMedicamentos;
222 int id_medicamento = claves_medicamentos[indice_medicamento];
223 int id_medic = medication[id_medicamento].get_id_num();
224 suministrarFarmacia(farmacia, id_medic, 10);
225 }
226 }
227 std::cout << "Cyclic medicine assignment completed." << std::endl;
228}
